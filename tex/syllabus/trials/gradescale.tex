\documentclass{article}
\makeatletter

%% GRADE SCALE

%% Set the percentage ranges for a letter grade (in csname array)
%% #1 Letter name, #2-4 Bottom value of minus, pure, and plus ranges
\newcommand{\GradeRange}[4]{%
	\expandafter\gdef\csname GradeRange#1\endcsname{#1 & #2 & #3 & #4}%
}
%% Recall array values
%% #1 Letter name
\newcommand{\@GetGradeRange}[1]{%
	\csname GradeRange#1\endcsname%
}
%% Print table of percentage values
\newcommand{\@PrintGradeScale}{%
	\begin{tabular}{lrrr}
		& \multicolumn{1}{c}{$-$} &  & \multicolumn{1}{c}{$+$}\\[1ex]
	\@GetGradeRange{A}\\
	\@GetGradeRange{B}\\
	\@GetGradeRange{C}\\
	\@GetGradeRange{D}\\
	\@GetGradeRange{F}\\
	\end{tabular}%
}
%% Wrap in environment and automatically print grade scale at end.
\newenvironment{GradeScale}
	{}
	{\medskip\@PrintGradeScale}

\makeatother

\begin{document}

\subsection{Grade Scale}

This table lists the minimum percentage points needed to earn each grade (90--92 is A$-$, 93--96 is A, 0--59 is F, etc.)

\begin{GradeScale}

\GradeRange{B}{80}{83}{87}
\GradeRange{C}{70}{73}{77}
\GradeRange{D}{}{60}{}
\GradeRange{F}{}{0}{}
\end{GradeScale}
