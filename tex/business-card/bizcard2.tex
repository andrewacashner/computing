\newcommand{\printgrid}[3]{% args: rows, columns, contents
    \newcount\row
    \newcount\col
    \row=#1
    \loop
        \col=#2
        {\loop
            #3%
            \advance\col by -1
            \ifnum\col > 0
        \repeat
        }
        \newline%
    \advance\row by -1
    \ifnum\row > 0
    \repeat
}
