% When I make the array automatically and print the results manually, it works.
\newcount\word
\word=0
\def\addtolist #1{
	\expandafter\def\csname word\the\word\endcsname{#1}
	\advance\word by 1
}
\addtolist{river}
\addtolist{r\'{\i}os}
