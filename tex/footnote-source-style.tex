  } % good
\sator.%
  \footnote{%
    Footnote with text of argument on separate line, but newline after brace commented out, and newline before closing brace commented out.%
  } % good, not worth the trouble
\sator.%

  \end{footnote} % good, not worth the trouble

  \end{Footnote} % could be valuable for long, multiparagraph notes
