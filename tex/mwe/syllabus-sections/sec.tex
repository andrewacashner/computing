\usepackage{titlesec}
\usepackage{titletoc}

\titleclass{\Unit}{straight}[\section]
\titleclass{\Week}{straight}[\subsection]
\titleclass{\Class}{straight}[\subsubsection]

\newcounter{Unit}
\newcounter{Week}
\newcounter{Class}

\setcounter{secnumdepth}{6}
\setcounter{tocdepth}{6}

\renewcommand{\theUnit}{\Roman{Unit}}
\renewcommand{\theWeek}{\arabic{Week}}
\renewcommand{\theClass}{\arabic{Class}}

\titleformat{\Unit}
	{\normalfont\Large\bfseries}
	{Unit \theUnit}
	{1em}
	{}
\titleformat{\Week}
	{\normalfont\large\bfseries}
	{Week \theWeek}
	{1em}
	{}
\titleformat{\Class}
	{\normalfont\bfseries}
	{Class \theClass}
	{1em}
	{}


\newlength{\secspacing}
\setlength{\secspacing}{3.25ex plus 1ex minus 0.2ex}
\titlespacing*{\Unit}{0pt}{\secspacing}{1em}
\titlespacing*{\Week}{0pt}{\secspacing}{1em}
\titlespacing*{\Class}{0pt}{\secspacing}{1em}

\makeatletter
\def\toclevel@Unit{1}
\def\toclevel@Week{2}
\def\toclevel@Class{3}

\newlength{\tocsep}
\setlength{\tocsep}{1.5em}
\newcommand*\l@Unit{\@dottedtocline{1}{\tocsep}{\tocsep}}
\newcommand*\l@Week{\@dottedtocline{2}{2\tocsep}{\tocsep}}
\newcommand*\l@Class{\@dottedtocline{3}{3\tocsep}{\tocsep}}
\makeatother
