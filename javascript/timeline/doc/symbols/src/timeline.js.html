 47 /**
 48  * We use this class to store information on the historical events used as
 49  * clues and inserted into the timeline, and to generate the HTML node for a
 50  * card (div.card). The HTML differs for clues vs. answers (clue shows "??" as
 51  * its date and is a draggable object; answer shows the real date and is a
 52  * drop target).
 53  *
 54  * @constructor
 55  * @param {number} date - Year of event 
 56  *      (NB - We cannot process more complex dates, just years)
 57  * @param {string} info - Brief description of event 
 58  */
 59 class Card {
 60     date;
 61     info;
 62 
 63     constructor(date, info) {
 64         this.date = date;
 65         this.info = info;
 66     }
 67     
 68     /**
 69      * Create HTML div.card node
 70      * @param {string} mode - "answer" or other ("question")
 71      *
 72      */
 73     toHtml(mode) {
 74         let card = document.createElement("div");
 75         card.className = "card";
 76         card.setAttribute("data-when", this.date);
 77 
 78         if (mode === "answer") {
 79             makeDropTarget(card);
 80         } else {
 81             makeDraggable(card);
 82         }
 83 
 84         let dateNode = document.createElement("span");
 85         dateNode.className = "date";
 86 
 87         if (mode === "answer") {
 88             dateNode.textContent = this.date;
 89         } else {
 90             dateNode.textContent = "??";
 91         }
 92 
 93         let infoNode = document.createElement("span");
 94         infoNode.className = "info";
 95         infoNode.textContent = this.info;
 96 
 97         card.appendChild(dateNode);
 98         card.appendChild(infoNode);
 99 
100         return card;
101     }
102 }

104 /** 
105  * Set card node as a drop target, not a draggable object.
106  * @param {Element} cardNode - a Card DOM Element (div.card)
107  */
108 function makeDropTarget(cardNode) {
109     cardNode.removeAttribute("draggable");
110     cardNode.removeAttribute("ondragstart");
111     cardNode.setAttribute("ondrop", "dropHandler(event)");
112     cardNode.setAttribute("ondragover", "dragoverHandler(event)");
113 }
114 
115 /** 
116  * Set card node as a draggable object, not a drop target.
117  * @param {Element} cardNode - a Card DOM Element (div.card)
118  */
119 function makeDraggable(cardNode) {
120     cardNode.removeAttribute("ondrop");
121     cardNode.removeAttribute("ondragover");
122     cardNode.setAttribute("draggable", "true");
123     cardNode.setAttribute("ondragstart", "dragstartHandler(event)");
124 }

126 /**
127  * Return a random integer up to the given max.
128  * @param {number} max
129  */
130 function randomInt(max) {
131     return Math.floor(Math.random() * max);
132 }
133 
134 /**
135  * Return a shuffled copy of a given array, using the Fisher-Yates/Knuth
136  * shuffle (`https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle`)
137  * @param {Array} array
138  */
139 function newShuffledArray(array) {
140 
141     function swapIndices(array, i, j) {
142         [array[i], array[j]] = [array[j], array[i]];
143         return array;
144     }
145 
146     let newArray = [...array];
147     for (let i = newArray.length - 1; i > 0; --i) {
148         newArray = swapIndices(newArray, i, randomInt(i));
149     }
150     return newArray;
151 }

153 class FactList {
154     constructor(factArray) {
155         let clueList = [];
156         for (let fact of factArray) {
157             let newClue = new Card(fact.date, fact.info);
158             clueList.push(newClue);
159         }
160         this.facts = newShuffledArray(clueList);
161     }
162 
163     extractEvent() {
164         let i = randomInt(this.facts.length);
165         let fact = this.facts[i];
166 
167         this.facts.splice(i, 1);
168         return fact;
169     }
170 
171     isEmpty() {
172         return this.facts.length === 0;
173     }
174 }

185 function drawNextClue(state) {
186     if (state.clues.isEmpty()) {
187         gameOver(state.score);
188     } else {
189         let clue = state.clues.extractEvent();
190         let clueNode = clue.toHtml("clue");
191         let clueBay = document.querySelector("div.clue");
192         clueBay.appendChild(clueNode);
193     }
194 }

214 function dragstartHandler(event) {
215     event.dataTransfer.setData("date", event.target.dataset.when);
216     event.dataTransfer.effectAllowed = "move";
217 }
218 
219 function dragoverHandler(event) {
220     event.preventDefault();
221     event.dataTransfer.effectAllowed = "move";

232 function isClueBetweenDates(clue, guess, preGuess) {
233     let clueDate = clue.dataset.when;
234     let guessDate = guess.dataset.when;
235     
236     let isBeforeGuess = clueDate <= guessDate;
237 
238     let noPreGuess = !preGuess;
239     let isAfterPreGuess = preGuess && (clueDate >= preGuess.dataset.when);
240     let isAfterAnyPreGuess = noPreGuess || isAfterPreGuess;
241 
242     return isBeforeGuess && isAfterAnyPreGuess;
243 }
